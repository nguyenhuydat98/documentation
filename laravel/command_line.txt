# Install package php7.2
	sudo apt-get install php7.2 libapache2-mod-php7.2 php7.2-common php7.2-gd php7.2-mysql php7.2-curl php7.2-intl php7.2-xsl php7.2-mbstring php7.2-zip php7.2-bcmath php7.2-soap php-xdebug php-imagick

# Tạo mới 1 project laravel
	composer create-project laravel/laravel <project-name>

# Kiểm tra phiên bản Laravel
	php artisan --version

# Xem danh sách route
	php artisan route:list

# Middleware
	php artisan make:middleware <middleware-name>

# Controller
	php artisan make:controller <controller-name>
	php artisan make:controller <controller-name> --resource

# Model
	php artisan make:model Product

# Migration
	php artisan migrate
	php artisan make:migration create_table_products_table --create=products
	php artisan make:migration change_column_image_to_image_path --table=products
	php artisan migrate:rollback
	php artisan migrate:rollback --step=n
	php artisan migrate:reset

	php artisan migrate:refresh

#Seeder and Factory
	php artisan make:factory <factory-name>
	php artisan make:seeder <seeder-name>
	php artisan db:seed
	php artisan db:seed --class=<class-seeder>
